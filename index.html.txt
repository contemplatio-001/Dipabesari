            saveToStorage() {

            setupEventListeners() {
                const container = document.getElementById('canvas-container');
                
                // Pan functionality
                container.addEventListener('mousedown', (e) => {
                    if (e.target === container || e.target.id === 'connections') {
                        this.isDragging = true;
                        this.lastX = e.clientX;
                        this.lastY = e.clientY;
                        container.style.cursor = 'grabbing';
                    }
                });

                window.addEventListener('mousemove', (e) => {
                    if (this.isDragging) {
                        const dx = e.clientX - this.lastX;
                        const dy = e.clientY - this.lastY;
                        this.translateX += dx;
                        this.translateY += dy;
                        this.lastX = e.clientX;
                        this.lastY = e.clientY;
                        this.updateTransform();
                    }
                });

                window.addEventListener('mouseup', () => {
                    this.isDragging = false;
                    container.style.cursor = 'grab';
                });

                // Zoom with wheel
                container.addEventListener('wheel', (e) => {
                    e.preventDefault();
                    const delta = e.deltaY > 0 ? 0.9 : 1.1;
                    this.scale = Math.max(0.3, Math.min(3, this.scale * delta));
                    this.updateTransform();
                });

                // Close context menu on click elsewhere
                window.addEventListener('click', (e) => {
                    if (!e.target.closest('#context-menu')) {
                        document.getElementById('context-menu').classList.add('hidden');
                    }
                });

                // Keyboard shortcuts
                window.addEventListener('keydown', (e) => {
                    if (e.key === 'Delete' && this.selectedNode) {
                        this.deleteMember();
                    }
                    if (e.key === 'Escape') {
                        this.closeModal();
                        document.getElementById('context-menu').classList.add('hidden');
                    }
                });
            }

            updateTransform() {
                const nodesContainer = document.getElementById('nodes-container');
                const connections = document.getElementById('connections');
                const transform = `translate(${this.translateX}px, ${this.translateY}px) scale(${this.scale})`;
                nodesContainer.style.transform = transform;
                
                // Update SVG viewBox for connections
                const rect = document.getElementById('canvas-container').getBoundingClientRect();
                connections.setAttribute('viewBox', `${-this.translateX} ${-this.translateY} ${rect.width / this.scale} ${rect.height / this.scale}`);
            }

            calculateLayout() {
                // Simple tree layout algorithm
                const levels = {};
                const visited = new Set();
                
                // Group by levels
                this.members.forEach(member => {
                    const level = this.getLevel(member.id);
                    if (!levels[level]) levels[level] = [];
                    levels[level].push(member);
                });

                // Position nodes
                const maxLevel = Math.max(...Object.keys(levels).map(Number));
                const centerX = window.innerWidth / 2;
                const startY = 100;

                for (let level = 0; level <= maxLevel; level++) {
                    const nodes = levels[level] || [];
                    const totalWidth = (nodes.length - 1) * this.siblingGap;
                    let startX = centerX - totalWidth / 2;

                    nodes.forEach((node, index) => {
                        // If has partner, position them together
                        if (node.partnerId && node.gender === 'male') {
                            node.x = startX + index * this.siblingGap;
                            const partner = this.members.find(m => m.id === node.partnerId);
                            if (partner) {
                                partner.x = node.x + 240;
                                partner.y = startY + level * this.levelHeight;
                            }
                        } else if (!node.partnerId || node.gender === 'female') {
                            // Check if already positioned as partner
                            const partner = this.members.find(m => m.partnerId === node.id);
                            if (!partner) {
                                node.x = startX + index * this.siblingGap;
                            }
                        }
                        node.y = startY + level * this.levelHeight;
                    });
                }
            }

            getLevel(memberId) {
                let level = 0;
                let current = this.members.find(m => m.id === memberId);
                while (current && current.parentId) {
                    level++;
                    current = this.members.find(m => m.id === current.parentId);
                }
                return level;
            }

            render() {
                this.calculateLayout();
                
                const container = document.getElementById('nodes-container');
                const svg = document.getElementById('connections');
                
                container.innerHTML = '';
                svg.innerHTML = '';

                // Draw connections first
                this.drawConnections(svg);
                
                // Draw nodes
                this.members.forEach(member => {
                    const node = this.createNodeElement(member);
                    container.appendChild(node);
                });

                this.updateTransform();
            }

            drawConnections(svg) {
                this.members.forEach(member => {
                    // Connection to parent
                    if (member.parentId) {
                        const parent = this.members.find(m => m.id === member.parentId);
                        if (parent) {
                            this.drawLine(svg, parent.x + 100, parent.y + 50, member.x + 100, member.y, 'parent');
                        }
                    }
                    
                    // Connection to partner
                    if (member.partnerId && member.gender === 'male') {
                        const partner = this.members.find(m => m.id === member.partnerId);
                        if (partner) {
                            this.drawLine(svg, member.x + 200, member.y + 50, partner.x, partner.y + 50, 'partner');
                        }
                    }
                });
            }

            drawLine(svg, x1, y1, x2, y2, type) {
                const path = document.createElementNS('http://www.w3.org/2000/svg', 'path');
                
                let d;
                if (type === 'partner') {
                    d = `M ${x1} ${y1} L ${x2} ${y2}`;
                } else {
                    // Curved line for parent-child
                    const midY = (y1 + y2) / 2;
                    d = `M ${x1} ${y1} C ${x1} ${midY}, ${x2} ${midY}, ${x2} ${y2}`;
                }
                
                path.setAttribute('d', d);
                path.setAttribute('class', 'connection-line');
                if (type === 'partner') {
                    path.style.strokeDasharray = '5,5';
                }
                svg.appendChild(path);
            }

            createNodeElement(member) {
                const div = document.createElement('div');
                div.className = `node-card absolute glass-panel rounded-2xl p-4 w-[200px] cursor-pointer border-2 ${member.gender === 'male' ? 'gender-male' : 'gender-female'} ${member.status === 'deceased' ? 'opacity-60' : ''}`;
                div.style.left = `${member.x}px`;
                div.style.top = `${member.y}px`;
                div.dataset.id = member.id;
                
                if (this.selectedNode === member.id) {
                    div.classList.add('ring-2', 'ring-purple-400', 'ring-offset-2', 'ring-offset-slate-900');
                }

                const photoUrl = member.photo || `https://i.pravatar.cc/150?u=${member.id}`;
                
                div.innerHTML = `
                    <div class="flex items-center gap-3">
                        <div class="relative">
                            <img src="${photoUrl}" alt="${member.name}" class="w-12 h-12 rounded-full object-cover border-2 ${member.gender === 'male' ? 'border-blue-400' : 'border-pink-400'}">
                            ${member.status === 'deceased' ? '<div class="absolute inset-0 rounded-full bg-black/50 flex items-center justify-center"><i class="fas fa-cross text-xs text-white"></i></div>' : ''}
                        </div>
                        <div class="flex-1 min-w-0">
                            <h3 class="font-semibold text-sm truncate">${member.name}</h3>
                            <p class="text-xs text-gray-400">${member.birthYear || '-'}</p>
                            <span class="inline-block mt-1 px-2 py-0.5 rounded-full text-[10px] ${member.gender === 'male' ? 'bg-blue-500/20 text-blue-300' : 'bg-pink-500/20 text-pink-300'}">
                                ${member.gender === 'male' ? 'L' : 'P'}
                            </span>
                        </div>
                    </div>
                `;

                // Events
                div.addEventListener('click', (e) => {
                    e.stopPropagation();
                    this.selectNode(member.id);
                });

                div.addEventListener('contextmenu', (e) => {
                    e.preventDefault();
                    this.selectNode(member.id);
                    this.showContextMenu(e.clientX, e.clientY);
                });

                div.addEventListener('mouseenter', (e) => {
                    this.showTooltip(e, member);
                });

                div.addEventListener('mouseleave', () => {
                    document.getElementById('tooltip').style.opacity = '0';
                });

                return div;
            }

            showTooltip(e, member) {
                const tooltip = document.getElementById('tooltip');
                tooltip.innerHTML = `
                    <div class="font-semibold">${member.name}</div>
                    <div class="text-gray-300">Lahir: ${member.birthYear || 'Unknown'}</div>
                    <div class="text-gray-300">Status: ${member.status === 'living' ? 'Hidup' : 'Wafat'}</div>
                    ${member.notes ? `<div class="mt-1 text-gray-400 italic">${member.notes}</div>` : ''}
                `;
                tooltip.style.left = e.pageX + 10 + 'px';
                tooltip.style.top = e.pageY + 10 + 'px';
                tooltip.style.opacity = '1';
            }

            selectNode(id) {
                this.selectedNode = id;
                this.render();
            }

            showContextMenu(x, y) {
                const menu = document.getElementById('context-menu');
                menu.style.left = Math.min(x, window.innerWidth - 200) + 'px';
                menu.style.top = Math.min(y, window.innerHeight - 150) + 'px';
                menu.classList.remove('hidden');
            }

            addRootMember() {
                document.getElementById('member-form').reset();
                document.getElementById('member-id').value = '';
                document.getElementById('parent-id').value = '';
                document.getElementById('modal-title').textContent = 'Tambah Anggota Root';
                document.getElementById('member-modal').classList.remove('hidden');
                document.getElementById('member-modal').classList.add('flex');
            }

            addChild() {
                if (!this.selectedNode) return;
                document.getElementById('member-form').reset();
                document.getElementById('member-id').value = '';
                document.getElementById('parent-id').value = this.selectedNode;
                document.getElementById('modal-title').textContent = 'Tambah Anak';
                document.getElementById('member-modal').classList.remove('hidden');
                document.getElementById('member-modal').classList.add('flex');
                document.getElementById('context-menu').classList.add('hidden');
            }

            addPartner() {
                if (!this.selectedNode) return;
                const member = this.members.find(m => m.id === this.selectedNode);
                if (member.partnerId) {
                    alert('Anggota ini sudah memiliki pasangan!');
                    return;
                }
                
                document.getElementById('member-form').reset();
                document.getElementById('member-id').value = '';
                document.getElementById('parent-id').value = '';
                document.getElementById('modal-title').textContent = 'Tambah Pasangan';
                document.getElementById('member-modal').classList.remove('hidden');
                document.getElementById('member-modal').classList.add('flex');
                document.getElementById('context-menu').classList.add('hidden');
                
                // Store temporary reference to set partner later
                this.tempPartnerFor = this.selectedNode;
            }

            editMember() {
                if (!this.selectedNode) return;
                const member = this.members.find(m => m.id === this.selectedNode);
                
                document.getElementById('member-id').value = member.id;
                document.getElementById('parent-id').value = member.parentId || '';
                document.getElementById('member-name').value = member.name;
                document.getElementById('member-gender').value = member.gender;
                document.getElementById('member-birth').value = member.birthYear || '';
                document.getElementById('member-status').value = member.status;
                document.getElementById('member-photo').value = member.photo || '';
                document.getElementById('member-notes').value = member.notes || '';
                document.getElementById('modal-title').textContent = 'Edit Anggota';
                
                document.getElementById('member-modal').classList.remove('hidden');
                document.getElementById('member-modal').classList.add('flex');
                document.getElementById('context-menu').classList.add('hidden');
            }

            deleteMember() {
                if (!this.selectedNode) return;
                if (!confirm('Apakah Anda yakin ingin menghapus anggota ini? Semua keturunannya juga akan terhapus.')) return;
                
                const toDelete = new Set();
                const queue = [this.selectedNode];
                
                while (queue.length > 0) {
                    const id = queue.shift();
                    toDelete.add(id);
                    
                    // Find children
                    this.members.filter(m => m.parentId === id).forEach(child => {
                        if (!toDelete.has(child.id)) queue.push(child.id);
                    });
                }
                
                this.members = this.members.filter(m => !toDelete.has(m.id));
                
                // Remove partner references
                this.members.forEach(m => {
                    if (toDelete.has(m.partnerId)) m.partnerId = null;
                });
                
                this.selectedNode = null;
                this.saveToStorage();
                this.render();
                this.updateStats();
                document.getElementById('context-menu').classList.add('hidden');
            }

            saveMember(e) {
                e.preventDefault();
                
                const id = document.getElementById('member-id').value;
                const parentId = document.getElementById('parent-id').value;
                const name = document.getElementById('member-name').value;
                const gender = document.getElementById('member-gender').value;
                const birthYear = document.getElementById('member-birth').value;
                const status = document.getElementById('member-status').value;
                const photo = document.getElementById('member-photo').value;
                const notes = document.getElementById('member-notes').value;
                
                if (id) {
                    // Update existing
                    const member = this.members.find(m => m.id === id);
                    member.name = name;
                    member.gender = gender;
                    member.birthYear = birthYear ? parseInt(birthYear) : null;
                    member.status = status;
                    member.photo = photo;
                    member.notes = notes;
                } else {
                    // Create new
                    const newMember = {
                        id: Date.now().toString(),
                        name,
                        gender,
                        birthYear: birthYear ? parseInt(birthYear) : null,
                        status,
                        photo,
                        notes,
                        parentId: parentId || null,
                        partnerId: null,
                        x: 0,
                        y: 0
                    };
                    
                    // If adding partner
                    if (this.tempPartnerFor) {
                        newMember.partnerId = this.tempPartnerFor;
                        const partner = this.members.find(m => m.id === this.tempPartnerFor);
                        if (partner) partner.partnerId = newMember.id;
                        this.tempPartnerFor = null;
                    }
                    
                    this.members.push(newMember);
                }
                
                this.saveToStorage();
                this.closeModal();
                this.render();
                this.updateStats();
            }

            closeModal() {
                document.getElementById('member-modal').classList.add('hidden');
                document.getElementById('member-modal').classList.remove('flex');
                this.tempPartnerFor = null;
            }

            zoomIn() {
                this.scale = Math.min(3, this.scale * 1.2);
                this.updateTransform();
            }

            zoomOut() {
                this.scale = Math.max(0.3, this.scale / 1.2);
                this.updateTransform();
            }

            resetView() {
                this.scale = 1;
                this.translateX = 0;
                this.translateY = 0;
                this.updateTransform();
            }

            updateStats() {
                document.getElementById('stat-total').textContent = this.members.length;
                
                const maxLevel = Math.max(...this.members.map(m => this.getLevel(m.id)), 0);
                document.getElementById('stat-generations').textContent = maxLevel + 1;
            }

            exportData() {
                const dataStr = JSON.stringify(this.members, null, 2);
                const blob = new Blob([dataStr], { type: 'application/json' });
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = `pohon-nasab-${new Date().toISOString().split('T')[0]}.json`;
                a.click();
                URL.revokeObjectURL(url);
            }

            importData(input) {
                const file = input.files[0];
                if (!file) return;
                
                const reader = new FileReader();
                reader.onload = (e) => {
                    try {
                        this.members = JSON.parse(e.target.result);
                        this.saveToStorage();
                        this.render();
                        this.updateStats();
                        alert('Data berhasil diimport!');
                    } catch (err) {
                        alert('File tidak valid!');
                    }
                };
                reader.readAsText(file);
                input.value = '';
            }
        }
